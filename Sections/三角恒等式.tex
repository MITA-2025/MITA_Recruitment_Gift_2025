\heading{三角恒等式}

\par 在下面的一切恒等式中, 我们要求 $\alpha$ 和 $\beta$ 使得对应的三角函数值有意义.

\Formula{基本恒等式}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin^{2}\alpha+\cos^{2}\alpha=1\quad,\quad\tan\alpha\cdot\cot\alpha=1\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan^{2}\alpha+1=\sec^{2}\alpha\quad,\quad\cot^{2}\alpha+1=\csc^{2}\alpha\ .
	\end{align*}
}

\Formula{诱导公式I}{
	$\FORALLII{\alpha}{\bC}{k}{\bZ}$
	\begin{align*}
		\sin\braI{2k\pi+\alpha}=\sin\alpha\quad,\quad\cos\braI{2k\pi+\alpha}=\cos\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan\braI{2k\pi+\alpha}=\tan\alpha\quad,\quad\cot\braI{2k\pi+\alpha}=\cot\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\sec\braI{2k\pi+\alpha}=\sec\alpha\quad,\quad\csc\braI{2k\pi+\alpha}=\csc\alpha\ .
	\end{align*}
}

\Formula{诱导公式II}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin\braI{-\alpha}=-\sin\alpha\quad,\quad\cos\braI{-\alpha}=\cos\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan\braI{-\alpha}=-\tan\alpha\quad,\quad\cot\braI{-\alpha}=-\cot\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\sec\braI{-\alpha}=-\sec\alpha\quad,\quad\csc\braI{-\alpha}=\csc\alpha\ .
	\end{align*}
}

\Formula{诱导公式III}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin\braI{\pi+\alpha}=-\sin\alpha\quad,\quad\cos\braI{\pi+\alpha}=-\cos\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan\braI{\pi+\alpha}=\tan\alpha\quad,\quad\cot\braI{\pi+\alpha}=\cot\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\sec\braI{\pi+\alpha}=-\sec\alpha\quad,\quad\csc\braI{\pi+\alpha}=-\csc\alpha\ .
	\end{align*}
}

\Formula{诱导公式IV}{

	\begin{align*}

\Formula{诱导公式V}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin\braI{\sfrac{\pi}{2}+\alpha}=\cos\alpha\quad,\quad\cos\braI{\sfrac{\pi}{2}+\alpha}=-\sin\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan\braI{\sfrac{\pi}{2}+\alpha}=-\cot\alpha\quad,\quad\cot\braI{\sfrac{\pi}{2}+\alpha}=-\tan\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\sec\braI{\sfrac{\pi}{2}+\alpha}=-\csc\alpha\quad,\quad\csc\braI{\sfrac{\pi}{2}+\alpha}=\sec\alpha\ .
	\end{align*}
}

\Formula{诱导公式VI}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin\braI{\sfrac{\pi}{2}-\alpha}=\cos\alpha\quad,\quad\cos\braI{\sfrac{\pi}{2}-\alpha}=\sin\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan\braI{\sfrac{\pi}{2}-\alpha}=\cot\alpha\quad,\quad\cot\braI{\sfrac{\pi}{2}-\alpha}=\tan\alpha\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\sec\braI{\sfrac{\pi}{2}-\alpha}=\csc\alpha\quad,\quad\csc\braI{\sfrac{\pi}{2}-\alpha}=\sec\alpha\ .
	\end{align*}
}

\Formula{和差公式}{
	$\FORALL{\alpha\and\beta}{\bC}$
	\begin{align*}
		\sin\braI{\alpha\pm\beta}=\sin\alpha\cos\beta\pm\cos\alpha\sin\beta\quad,\quad\cos\braI{\alpha\pm\beta}=\cos\alpha\cos\beta\mp\sin\alpha\sin\beta\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\tan\braI{\alpha\pm\beta}=\frac{\tan\alpha\pm\tan\beta}{1\mp\tan\alpha\tan\beta}\quad,\quad\cot\braI{\alpha\pm\beta}=\frac{\cot\alpha\cot\beta\mp1}{\cot\alpha\pm\cot\beta}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\sec\braI{\alpha\pm\beta}=\frac{\sec\alpha\sec\beta}{1\mp\tan\alpha\tan\beta}\quad,\quad\csc\braI{\alpha\pm\beta}=\frac{\sec\alpha\sec\beta}{\tan\alpha\pm\tan\beta}\ .
	\end{align*}
}

\Formula{辅助角公式}{
	$\FORALLII{\alpha}{\bC}{A\and B}{\bC}$
	\begin{align*}
		A\sin\alpha+B\cos\alpha&=\sqrt{\dis A^{2}+B^{2}}\sin\braI{\alpha+\varphi}\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		A\sin^{2}\alpha+B\cos^{2}\alpha+C\sin\alpha\cos\alpha=\frac{\sqrt{\dis A^{2}+B^{2}+C^{2}-2AB}}{2}\sin\braI{2\alpha+\psi}+\frac{A+B}{2}\ ,
	\end{align*}
	其中辅助角 $\varphi$ 和 $\psi$ 满足
	\begin{align*}
		\braI{\cos\varphi\and\sin\varphi}=\braI{A\and B}\AND\braI{\cos\psi\and\sin\psi}=\braI{C\and B-A}\ .
	\end{align*}
}

\Formula{半角公式}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\tan\frac{\alpha}{2}=\frac{\sin\alpha}{1+\cos\alpha}=\frac{1-\cos\alpha}{\sin\alpha}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cot\frac{\alpha}{2}=\frac{\sin\alpha}{1-\cos\alpha}=\frac{1+\cos\alpha}{\sin\alpha}\ .
	\end{align*}
}

\Formula{倍角公式}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin2\alpha=2\sin\alpha\cos\alpha\quad,\quad\cos2\alpha=\cos^{2}\alpha-\sin^{2}\alpha\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\tan2\alpha=\frac{2\tan\alpha}{\dis1-\tan^{2}\alpha}\quad,\quad\cot2\alpha=\frac{\cot^{2}\alpha-1}{2\cot\alpha}\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\sin3\alpha=-4\sin^{3}\alpha+3\sin\alpha\quad,\quad\cos3\alpha=4\cos^{3}\alpha-3\cos\alpha\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\tan3\alpha=\frac{\dis\tan^{3}\alpha-3\tan\alpha}{\dis3\tan^{2}\alpha-1}\quad,\quad\cot3\alpha=\frac{\cot^{3}\alpha-3\cot\alpha}{3\cot^{2}\alpha-1}\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\sin4\alpha=\cos\alpha\braI{-8\sin^{3}\alpha+4\sin\alpha}\quad,\quad\cos4\alpha=8\cos^{4}\alpha-8\cos^{2}\alpha+1\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\tan4\alpha=\frac{\dis-4\tan^{3}\alpha+4\tan\alpha}{\dis\tan^{4}\alpha-6\tan^{2}\alpha+1}\quad,\quad\cot4\alpha=\frac{\cot^{4}\alpha-6\cot^{2}\alpha+1}{4\cot^{3}\alpha-4\cot\alpha}\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\sin5\alpha=16\sin^{5}\alpha-20\sin^{3}\alpha+5\sin\alpha\quad,\quad\cos5\alpha=16\cos^{5}\alpha-20\cos^{3}\alpha+5\cos\alpha\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\tan5\alpha=\frac{\dis\tan^{5}\alpha-10\tan^{3}\alpha+5\tan\alpha}{\dis5\tan^{4}\alpha-10\tan^{2}\alpha+1}\quad,\quad\cot5\alpha=\frac{\dis\cot^{5}\alpha-10\cot^{3}\alpha+5\cot\alpha}{\dis5\cot^{4}\alpha-10\cot^{2}\alpha+1}\ .
	\end{align*}
	$\FORALLII{\alpha}{\bC}{n}{\bN}$
	\begin{align*}
		\cos n\alpha=n\sum_{k=0}^{\left\lfloor\tfrac{n}{2}\right\rfloor}\braI{-1}^{k}\frac{\braI{2n-2k-2}\,!!}{\braI{2k}\,!!\braI{n-2k}\,!}\cos^{n-2k}\alpha\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\sin2n\alpha=\cos\alpha\sum_{k=0}^{n-1}\braI{-1}^{n-k-1}\frac{\braI{4n-2k-2}\,!!}{\braI{2k}\,!!\braI{2n-2k-1}\,!}\sin^{2n-2k-1}\alpha\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\sin\braI{2n+1}\alpha=\braI{2n+1}\sum_{k=0}^{n}\braI{-1}^{n-k}\frac{\braI{4n-2k}\,!!}{\braI{2k}\,!!\braI{2n-2k+1}\,!}\sin^{2n-2k+1}\alpha\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\tan n\alpha=\frac{\dis\sum^{\textstyle\braVI{\!\frac{n-1}{2}\!}}_{k=0}\braI{-1}^{k}\rC^{2k+1}_{n}\tan^{2k+1}\alpha}{\dis\sum^{\textstyle\braVI{\!\frac{n}{2}\!}}_{k=0}\braI{-1}^{k}\rC^{2k}_{n}\tan^{2k}\alpha}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cot n\alpha=\frac{\dis\sum^{\textstyle\braVI{\!\frac{n}{2}\!}}_{k=0}\braI{-1}^{k}\rC^{2k}_{n}\cot^{n-2k}\alpha}{\dis\sum^{\textstyle\braVI{\!\frac{n-1}{2}\!}}_{k=0}\braI{-1}^{k}\rC^{2k+1}_{n}\cot^{n-2k-1}\alpha}\ .
	\end{align*}
}

\Formula{降幂公式}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin^{2}\alpha=\frac{1-\cos2\alpha}{2}\quad,\quad\cos^{2}\alpha=\frac{1+\cos2\alpha}{2}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\tan^{2}\alpha=\frac{1-\cos2\alpha}{1+\cos2\alpha}\quad,\quad\cot^{2}\alpha=\frac{1+\cos2\alpha}{1-\cos2\alpha}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\sec^{2}\alpha=\frac{2}{1+\cos2\alpha}\quad,\quad\csc^{2}\alpha=\frac{2}{1-\cos2\alpha}\ .
	\end{align*}
	$\FORALLII{\alpha}{\bC}{n}{\bN}$
	\begin{align*}
	\sin^{2n+1}\alpha=\frac{1}{\braI{-4}^{n}}\sum^{n}_{k=0}\braI{-1}^{k}\rC^{k}_{2n+1}\sin\braI{2n-2k+1}\alpha\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
	\cos^{2n+1}\alpha=\frac{1}{4^{n}}\sum^{n}_{k=0}\rC^{k}_{2n+1}\cos\braI{2n-2k+1}\alpha\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
	\sin^{2n}\alpha=\frac{\braI{2n}!}{4^{n}\braI{n!}^{2}}+\frac{2}{\braI{-4}^{n}}\sum^{n-1}_{k=0}\braI{-1}^{k}\rC^{k}_{2n}\cos\braI{2n-2k}\alpha\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
	\cos^{2n}\alpha=\frac{\braI{2n}!}{4^{n}\braI{n!}^{2}}+\frac{2}{4^{n}}\sum^{n-1}_{k=0}\rC^{k}_{2n}\cos\left(2n-2k\right)\alpha\ .
	\end{align*}
}

\Formula{两元积化和差公式}{
	$\FORALL{\alpha\and\beta}{\bC}$
	\begin{align*}
		\sin\alpha\cos\beta=\frac{1}{2}\braII{\sin\braI{\alpha+\beta}+\sin\braI{\alpha-\beta}\!}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cos\alpha\cos\beta=\frac{1}{2}\braII{\cos\braI{\alpha+\beta}+\cos\braI{\alpha-\beta}\!}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\sin\alpha\sin\beta=-\frac{1}{2}\braII{\cos\braI{\alpha+\beta}-\cos\braI{\alpha-\beta}\!}\ .
	\end{align*}
}

\Formula{两元和差化积公式}{
	$\FORALL{\alpha\and\beta}{\bC}$
	\begin{align*}
		\sin\alpha\pm\sin\beta=2\sin\frac{\alpha\pm\beta}{2}\cos\frac{\alpha\mp\beta}{2}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cos\alpha+\cos\beta=2\cos\frac{\alpha+\beta}{2}\cos\frac{\alpha-\beta}{2}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cos\alpha-\cos\beta=-2\sin\frac{\alpha+\beta}{2}\sin\frac{\alpha-\beta}{2}\ .
	\end{align*}
}

\Formula{万能代换公式}{
	$\FORALL{\alpha}{\bC}$
	\begin{align*}
		\sin\alpha=\frac{2\tan\braI{\sfrac{\alpha}{2}}}{1+\tan^{2}\braI{\sfrac{\alpha}{2}}}\quad,\quad\cos\alpha=\frac{1-\tan^{2}\braI{\sfrac{\alpha}{2}}}{1+\tan^{2}\braI{\sfrac{\alpha}{2}}}\quad,\quad\tan\alpha=\frac{2\tan\braI{\sfrac{\alpha}{2}}}{1-\tan^{2}\braI{\sfrac{\alpha}{2}}}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cot\alpha=\frac{1-\tan^{2}\braI{\sfrac{\alpha}{2}}}{2\tan\braI{\sfrac{\alpha}{2}}}\quad,\quad\sec\alpha=\frac{1+\tan^{2}\braI{\sfrac{\alpha}{2}}}{1-\tan^{2}\braI{\sfrac{\alpha}{2}}}\quad,\quad\csc\alpha=\frac{1+\tan^{2}\braI{\sfrac{\alpha}{2}}}{2\tan\braI{\sfrac{\alpha}{2}}}\ .
	\end{align*}
}

\Formula{三角平方差公式}{
	$\FORALL{\alpha\and\beta}{\bC}$
	\begin{align*}
		\sin^{2}\alpha-\sin^{2}\beta=\sin\braI{\alpha+\beta}\sin\braI{\alpha-\beta}\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\cos^{2}\alpha-\sin^{2}\beta=\cos\braI{\alpha+\beta}\cos\braI{\alpha-\beta}\ .
	\end{align*}
}

\Formula{切化弦公式}{
	$\FORALL{\alpha\and\beta}{\bC}$
	\begin{align*}
		\tan\alpha\pm\tan\beta=\frac{\sin\braI{\alpha\pm\beta}}{\cos\alpha\cos\beta}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cot\alpha\pm\cot\beta=\frac{\sin\braI{\beta\pm\alpha}}{\sin\alpha\sin\beta}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\tan\alpha\pm\cot\beta=\pm\frac{\cos\braI{\beta\mp\alpha}}{\sin\beta\cos\alpha}\ .
	\end{align*}
}

\Formula{内角恒等式}{
	$\FORALLIF{\alpha\and\beta\and\gamma}{\bC}{\alpha+\beta+\gamma=\pi}$
	\begin{align*}
		\sin\alpha+\sin\beta+\sin\gamma=4\cos\frac{\alpha}{2}\cos\frac{\beta}{2}\cos\frac{\gamma}{2}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\cos\alpha+\cos\beta+\cos\gamma=4\sin\frac{\alpha}{2}\sin\frac{\beta}{2}\sin\frac{\gamma}{2}+1\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\sin2\alpha+\sin2\beta+\sin2\gamma=4\sin\alpha\sin\beta\sin\gamma\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\tan\alpha+\tan\beta+\tan\gamma=\tan\alpha\tan\beta\tan\gamma\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\cot\frac{\alpha}{2}+\cot\frac{\beta}{2}+\cot\frac{\gamma}{2}=\cot\frac{\alpha}{2}\cot\frac{\beta}{2}\cot\frac{\gamma}{2}\ ,\\[-13mm]
	\end{align*}
	\begin{align*}
		\tan\frac{\alpha}{2}\tan\frac{\beta}{2}+\tan\frac{\beta}{2}\tan\frac{\gamma}{2}+\tan\frac{\gamma}{2}\tan\frac{\alpha}{2}=1\ ,\\[-14mm]
	\end{align*}
	\begin{align*}
		\cot\alpha\cot\beta+\cot\beta\cot\gamma+\cot\gamma\cot\alpha=1\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\sin^{2}\alpha+\sin^{2}\beta+\sin^{2}\gamma=2+2\cos\alpha\cos\beta\cos\gamma\ ,\\[-15mm]
	\end{align*}
	\begin{align*}
		\cos^{2}\alpha+\cos^{2}\beta+\cos^{2}\gamma=1-2\cos\alpha\cos\beta\cos\gamma\ .
	\end{align*}
}
