\heading{初等函数导数表}

\Formula{幂函数}{
    $\FORALL{\mu}{\bR}$
    \begin{align*}
        \braI{x}\der=1\quad,\quad\braI{x^{\mu}}\der=\mu x^{\mu-1}\ .
    \end{align*}
}

\Formula{指数函数&对数函数}<指数函数\&对数函数>{
	$\FORALL{a}{\braI{0\,,\,+\infty}\setminus\braIII{1}}$
    \begin{align*}
        \braI{a^{x}}\der=a^{x}\ln a\quad,\quad\braI{\log_{\,a\!}x}\der=\frac{1}{x\ln a}\ .
    \end{align*}
}

\Formula{三角函数}{
    \begin{align*}
        \braI{\sin x}\der=\cos x\quad,\quad\braI{\cos x}\der=-\sin x\quad,\quad\braI{\tan x}\der=\sec^{2}x\ ,\\[-15mm]
    \end{align*}
    \begin{align*}
        \braI{\sec x}\der=\tan x\sec x\quad,\quad\braI{\csc x}\der=-\cot x\csc x\quad,\quad\braI{\cot x}\der=-\csc^{2}x\ .
    \end{align*}
}

\Formula{反三角函数}{
    \begin{align*}
        \braI{\arcsin x}\der=\frac{1}{\sqrt{1-x^{2}}}\quad,\quad\braI{\arccos x}\der=-\frac{1}{\sqrt{1-x^{2}}}\quad,\quad\braI{\arctan x}\der=\frac{1}{x^{2}+1}\ ,\\[-13mm]
    \end{align*}
    \begin{align*}
        \braI{\arcsec x}\der=\frac{1}{\braIV{x}\sqrt{x^{2}-1}}\quad,\quad\braI{\arccsc x}\der=-\frac{1}{\braIV{x}\sqrt{x^{2}-1}}\quad,\quad\braI{\arccot x}\der=-\frac{1}{x^{2}+1}\ .
    \end{align*}
}

\Formula{双曲函数}{
    \begin{align*}
        \braI{\sinh x}\der=\cosh x\quad,\quad\braI{\cosh x}\der=\sinh x\quad,\quad\braI{\tanh x}\der=\mathrm{sech}^{2}\,x\ ,\\[-15mm]
    \end{align*}
    \begin{align*}
        \braI{\sech x}\der=-\tanh x\,\sech x\quad,\quad\braI{\csch x}\der=-\coth x\,\csch x\quad,\quad\braI{\coth x}\der=-\mathrm{csch}^{2}\,x\ .
    \end{align*}
}

\Formula{反双曲函数}{
        \begin{align*}
            \braI{\arsinh x}\der=\frac{1}{\sqrt{x^{2}+1}}\quad,\quad\braI{\arcosh x}\der=\frac{1}{\sqrt{x^{2}-1}}\quad,\quad\braI{\artanh x}\der=\frac{1}{1-x^{2}}\ ,\\[-13mm]
        \end{align*}
        \begin{align*}
            \braI{\arsech x}\der=-\frac{1}{x\sqrt{1-x^{2}}}\quad,\quad\braI{\arcsch x}\der=-\frac{1}{\braIV{x}\sqrt{x^{2}+1}}\quad,\quad\braI{\arcoth x}\der=\frac{1}{1-x^{2}}\ .
        \end{align*}
}
